الخطوة 1: إنشاء نظام التزامن الموحد

// نظام التزامن الموحد الجديد class UnifiedSyncSystem { constructor() { this.STORAGE_KEYS = { DAILY_QUESTIONS: 'nafis_unified_daily_questions', WEEKLY_QUESTIONS: 'nafis_unified_weekly_questions', LAST_UPDATE: 'nafis_unified_last_update', SYNC_STATUS: 'nafis_unified_sync_status', HEARTBEAT: 'nafis_unified_heartbeat' }; this.channels = []; this.retryCount = 0; this.maxRetries = 5; this.heartbeatInterval = null; this.listeners = []; this.init(); } init() { this.setupChannels(); this.setupHeartbeat(); this.setupRetryMechanism(); this.bindStorageEvents(); console.log('✅ نظام التزامن الموحد جاهز'); } // إعداد قنوات الاتصال المتعددة setupChannels() { // BroadcastChannel if (typeof BroadcastChannel !== 'undefined') { const channel = new BroadcastChannel('nafis-unified-sync'); channel.onmessage = (e) => this.handleMessage(e.data); this.channels.push(channel); } // PostMessage window.addEventListener('message', (e) => { if (e.data && e.data.source === 'nafis-sync') { this.handleMessage(e.data); } }); // Storage Events window.addEventListener('storage', (e) => { if (e.key && e.key.startsWith('nafis_unified_')) { this.handleStorageChange(e); } }); } // آلية heartbeat للتأكد من الاتصال setupHeartbeat() { this.heartbeatInterval = setInterval(() => { const heartbeat = { timestamp: Date.now(), source: window.location.pathname, questions: { daily: this.getQuestionCount('daily'), weekly: this.getQuestionCount('weekly') } }; localStorage.setItem(this.STORAGE_KEYS.HEARTBEAT, JSON.stringify(heartbeat)); this.broadcast({ type: 'HEARTBEAT', data: heartbeat }); }, 3000); } // نشر البيانات عبر جميع القنوات broadcast(message) { const payload = { ...message, source: 'nafis-sync', timestamp: Date.now(), id: Math.random().toString(36).substr(2, 9) }; // BroadcastChannel this.channels.forEach(channel => { try { channel.postMessage(payload); } catch (e) { console.warn('فشل في إرسال عبر BroadcastChannel:', e); } }); // PostMessage try { window.postMessage(payload, '*'); } catch (e) { console.warn('فشل في إرسال عبر PostMessage:', e); } // Storage Event try { const storageKey = `nafis_temp_${payload.id}`; localStorage.setItem(storageKey, JSON.stringify(payload)); localStorage.removeItem(storageKey); } catch (e) { console.warn('فشل في إرسال عبر Storage:', e); } } // حفظ الأسئلة مع ضمانات الموثوقية saveQuestions(type, questions) { try { const key = type === 'daily' ? this.STORAGE_KEYS.DAILY_QUESTIONS : this.STORAGE_KEYS.WEEKLY_QUESTIONS; const data = { questions: questions, timestamp: Date.now(), checksum: this.calculateChecksum(questions), version: '2.0' }; // حفظ محلي localStorage.setItem(key, JSON.stringify(data)); // حفظ في Firebase if (window.firebaseDB) { this.saveToFirebase(type, data); } // تحديث الحالة localStorage.setItem(this.STORAGE_KEYS.LAST_UPDATE, Date.now().toString()); localStorage.setItem(this.STORAGE_KEYS.SYNC_STATUS, 'success'); // إشعار جميع المستمعين this.broadcast({ type: 'QUESTIONS_UPDATED', questionType: type, data: data, count: questions.length }); console.log(`✅ تم حفظ ${questions.length} سؤال ${type}`); return true; } catch (error) { console.error('❌ خطأ في حفظ الأسئلة:', error); this.handleError(error); return false; } } // تحميل الأسئلة مع التحقق من السلامة loadQuestions(type) { try { const key = type === 'daily' ? this.STORAGE_KEYS.DAILY_QUESTIONS : this.STORAGE_KEYS.WEEKLY_QUESTIONS; const stored = localStorage.getItem(key); if (!stored) return []; const data = JSON.parse(stored); // التحقق من سلامة البيانات if (this.validateData(data)) { return data.questions || []; } else { console.warn('⚠️ بيانات تالفة، محاولة الاستعادة من Firebase'); this.recoverFromFirebase(type); return []; } } catch (error) { console.error('❌ خطأ في تحميل الأسئلة:', error); return []; } } // حساب checksum للتحقق من سلامة البيانات calculateChecksum(data) { const str = JSON.stringify(data); let hash = 0; for (let i = 0; i < str.length; i++) { const char = str.charCodeAt(i); hash = ((hash << 5) - hash) + char; hash = hash & hash; // تحويل إلى 32bit integer } return hash.toString(); } // التحقق من صحة البيانات validateData(data) { if (!data || !data.questions || !Array.isArray(data.questions)) { return false; } if (data.checksum && this.calculateChecksum(data.questions) !== data.checksum) { return false; } return true; } // معالجة الرسائل الواردة handleMessage(message) { if (!message || message.source !== 'nafis-sync') return; switch (message.type) { case 'QUESTIONS_UPDATED': this.handleQuestionsUpdate(message); break; case 'HEARTBEAT': this.handleHeartbeat(message); break; case 'SYNC_REQUEST': this.handleSyncRequest(message); break; } } // معالجة تحديث الأسئلة handleQuestionsUpdate(message) { const { questionType, data } = message; if (this.validateData(data)) { const key = questionType === 'daily' ? this.STORAGE_KEYS.DAILY_QUESTIONS : this.STORAGE_KEYS.WEEKLY_QUESTIONS; localStorage.setItem(key, JSON.stringify(data)); // إشعار المستمعين المحليين this.notifyListeners('questions_updated', { type: questionType, questions: data.questions, count: data.questions.length }); console.log(`🔄 تم استقبال تحديث ${questionType}: ${data.questions.length} سؤال`); } } // إضافة مستمع للأحداث addListener(callback) { this.listeners.push(callback); } // إشعار المستمعين notifyListeners(event, data) { this.listeners.forEach(callback => { try { callback(event, data); } catch (e) { console.warn('خطأ في مستمع الأحداث:', e); } }); } // آلية إعادة المحاولة setupRetryMechanism() { setInterval(() => { const lastHeartbeat = localStorage.getItem(this.STORAGE_KEYS.HEARTBEAT); if (lastHeartbeat) { const heartbeat = JSON.parse(lastHeartbeat); const timeDiff = Date.now() - heartbeat.timestamp; if (timeDiff > 10000) { // 10 ثوان console.log('🔄 انقطاع الاتصال، محاولة إعادة التزامن...'); this.forceSync(); } } }, 5000); } // فرض التزامن forceSync() { this.broadcast({ type: 'SYNC_REQUEST', timestamp: Date.now() }); } // الحصول على عدد الأسئلة getQuestionCount(type) { const questions = this.loadQuestions(type); return questions.length; } // تنظيف الموارد destroy() { if (this.heartbeatInterval) { clearInterval(this.heartbeatInterval); } this.channels.forEach(channel => { try { channel.close(); } catch (e) { // تجاهل الأخطاء } }); this.listeners = []; console.log('🔄 تم تنظيف نظام التزامن'); } } // إنشاء مثيل النظام الموحد window.unifiedSync = new UnifiedSyncSystem();

الخطوة 3: تحديث منصة الطلاب

// إضافة هذا الكود إلى منصة الطلاب class StudentSyncManager { constructor() { this.dailyQuestions = []; this.weeklyQuestions = []; this.init(); } init() { // تحميل الأسئلة من النظام الموحد this.loadQuestions(); // الاستماع للتحديثات window.unifiedSync.addListener((event, data) => { if (event === 'questions_updated') { this.handleQuestionsUpdate(data); } }); // تحديث دوري setInterval(() => { this.loadQuestions(); }, 5000); } loadQuestions() { const newDaily = window.unifiedSync.loadQuestions('daily'); const newWeekly = window.unifiedSync.loadQuestions('weekly'); if (newDaily.length !== this.dailyQuestions.length || newWeekly.length !== this.weeklyQuestions.length) { this.dailyQuestions = newDaily; this.weeklyQuestions = newWeekly; this.updateUI(); this.showUpdateNotification(); } } updateUI() { const dailyElement = document.getElementById('dailyQuestionCount'); const weeklyElement = document.getElementById('weeklyQuestionCount'); if (dailyElement) { dailyElement.textContent = this.dailyQuestions.length > 0 ? `${this.dailyQuestions.length} أسئلة متنوعة` : 'لا توجد أسئلة متاحة'; } if (weeklyElement) { weeklyElement.textContent = this.weeklyQuestions.length > 0 ? `${this.weeklyQuestions.length} سؤال شامل` : 'لا توجد أسئلة متاحة'; } // تحديث مؤشر التزامن document.getElementById('syncStatus').innerHTML = '
محدث'; } handleQuestionsUpdate(data) { console.log(`🔄 تحديث تلقائي: ${data.type} - ${data.count} سؤال`); this.loadQuestions(); } showUpdateNotification() { const notification = document.createElement('div'); notification.className = 'notification'; notification.innerHTML = ` ✅
تم تحديث الأسئلة!
يومي: ${this.dailyQuestions.length} | أسبوعي: ${this.weeklyQuestions.length}
`; document.body.appendChild(notification); setTimeout(() => notification.remove(), 4000); } } // تفعيل مدير التزامن للطلاب window.dailyQuestions = []; window.weeklyQuestions = []; window.studentSync = new StudentSyncManager();
